Visualization #1!

This data represents the obesity rate per state.

From this, we can determine the state with the highest obesity rate is Indiana, and the lowest is Colorado. (woohoo go Colorado)

Shown below is the code used to create the previous visualization. In vis #1, I was focused on changing the color of the chart itself.

< div id="obesity-bars">< /div>
						
< script type="text/javascript">
var spec = {
"$schema": "https://vega.github.io/schema/vega-lite/v5.json",
	"width": 1000,
	"height": 300,
"config":{
	"font":"garamond"
},
"data": {
	"url": "https://vega.github.io/vega-datasets/data/obesity.json"
},
"mark": {
	"type":"bar",
	"color":"pink"
},
"encoding": {
	"x": {
		"field": "state",
		"type": "nominal",
		"title": "State",
	"axis": {
		"titleFontSize": 15
	}
},
	"y": {
		"field": "rate",
		"type": "quantitative",
		"title": "Rate",
	"axis": {
"		titleFontSize": 15
			}
		}
	}
};
var opt = {"renderer": "canvas", "actions": false};
vegaEmbed("#obesity-bars", spec, opt);
< /script>

Visualization #2

This dataset represents the relationship of penguin species to their body mass and flipper length.
From this visualization, we can see that most penguin species have a body mass over 5,000 g and flipper length over 150 mm.
This was the code used to create the previous visualization. In visualization #2, I was focused on creating a compostion in with the data.

< div id="penguin-vis">
					
< script type="text/javascript">
var spec = { 
"$schema": "https://vega.github.io/schema/vega-lite/v5.json",
"config": {
"view": {
"continuousHeight": 300,
"continuousWidth": 400
	}
},
"data": {
"url": "https://cdn.jsdelivr.net/npm/vega-datasets@2.2.0/data/penguins.json"
},
"mark": {
"type":"bar",
"color":"pink"
},
"encoding": {
"x": {
"field": "Body Mass (g)",
"type": "quantitative",
"bin": {
"maxbins": 15
},
"spacing": 10
},
"y": {
"aggregate": "count", 
"title": "Flipper Length (mm)",
"type": "quantitative"
},
"column": {
"field": "Species",
"type": "nominal"
}
}
};
var opt = {"renderer": "canvas", "actions": false};  
vegaEmbed("#penguin-vis", spec, opt);
< /script>
