Using Vega visualizations in reveal.js presentations!

How can we add these visualizations?

Let's get started

First step, you must add the tags to import Vega.


							{
							  "$schema": "https://vega.github.io/schema/vega-lite/v4.json",
							  "config": {
							    "view": {
							      "continuousHeight": 300,
							      "continuousWidth": 400
							    }
							  },
							  "data": {
							    "url": "https://vega.github.io/vega-datasets/data/cars.json"
							  },
							  "encoding": {
							    "color": {
							      "field": "Origin",
							      "type": "nominal"
							    },
							    "x": {
							      "field": "Horsepower",
							      "type": "quantitative"
							    },
							    "y": {
							      "field": "Miles_per_Gallon",
							      "type": "quantitative"
							    }
							  },
							  "mark": "point"
							};

Next, you add the Vega spec document into the "var spec = {}" line of code from before.


							{
							  "$schema": "https://vega.github.io/schema/vega-lite/v4.json",
							  "config": {
							    "view": {
							      "continuousHeight": 300,
							      "continuousWidth": 400
							    }
							  },
							  "data": {
							    "url": "https://vega.github.io/vega-datasets/data/cars.json"
							  },
							  "encoding": {
							    "color": {
							      "field": "Origin",
							      "type": "nominal"
							    },
							    "x": {
							      "field": "Horsepower",
							      "type": "quantitative"
							    },
							    "y": {
							      "field": "Miles_per_Gallon",
							      "type": "quantitative"
							    }
							  },
							  "mark": "point"
							};
